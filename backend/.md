import { useState, useEffect, useCallback } from "react";
import axios from "axios";
import Chat from "../components/Chat";

const RequestsInbox = () => {
  const [requests, setRequests] = useState([]);
  const [loading, setLoading] = useState(true);
  const [activeChat, setActiveChat] = useState(null);
  const user = JSON.parse(localStorage.getItem("user"));

  const fetchInbox = useCallback(async (isMounted) => {
    if (!user?._id) return;
    try {
      const res = await axios.get(`http://localhost:5000/api/requests/my-inbox/${user._id}`);
      if (isMounted) {
        setRequests(res.data);
        setLoading(false);
      }
    } catch (err) {
      if (isMounted) {
        console.error("Fetch Error:", err);
        setLoading(false);
      }
    }
  }, [user?._id]);

  useEffect(() => {
    let isMounted = true;
    
    // Async wrapper to avoid synchronous setState cascading renders
    const loadData = async () => {
      await fetchInbox(isMounted);
    };

    loadData();

    return () => {
      isMounted = false;
    };
  }, [fetchInbox]);

  const handleAction = async (id, status) => {
    try {
      const res = await axios.put(`http://localhost:5000/api/requests/update/${id}`, { status });
      
      if (status === "accepted" && res.data.updatedUser) {
        localStorage.setItem("user", JSON.stringify(res.data.updatedUser));
        window.dispatchEvent(new Event("storage")); 
      }
      
      // Update local state: remove if rejected, reload if accepted to refresh views
      if (status === "rejected") {
        setRequests((prev) => prev.filter((r) => r._id !== id));
      } else {
        window.location.reload(); 
      }
    } catch (err) {
      console.error("Action Error:", err);
    }
  };

  if (loading) return <div className="p-10 text-center font-black animate-pulse">Syncing...</div>;

  return (
    <div className="min-h-screen bg-gray-50 p-8 flex gap-6">
      <div className="flex-grow max-w-4xl">
        <h1 className="text-3xl font-black mb-8 text-gray-900">SkillSwap Inbox</h1>
        <div className="grid gap-4">
          {requests.map((req) => {
            const isTeacher = req.receiverId?._id === user._id;
            const partner = isTeacher ? req.senderId : req.receiverId;

            return (
              <div key={req._id} className="bg-white p-6 rounded-3xl border border-gray-100 shadow-sm flex justify-between items-center">
                <div>
                  <h3 className="text-xl font-bold text-gray-800">{partner?.name}</h3>
                  <p className="text-gray-500 font-medium">Skill: {req.skillName} | Status: <span className="font-bold text-blue-600 uppercase">{req.status}</span></p>
                </div>
                
                <div className="flex gap-2">
                  {req.status === "pending" && isTeacher && (
                    <>
                      <button onClick={() => handleAction(req._id, "accepted")} className="bg-blue-600 text-white px-6 py-2 rounded-xl font-bold">Accept</button>
                      <button onClick={() => handleAction(req._id, "rejected")} className="bg-red-50 text-red-600 px-6 py-2 rounded-xl font-bold">Decline</button>
                    </>
                  )}

                  {req.status === "accepted" && (
                    <button 
                      onClick={() => setActiveChat({ id: partner?._id, name: partner?.name })}
                      className="bg-green-600 text-white px-6 py-2 rounded-xl font-bold shadow-lg shadow-green-100"
                    >
                      💬 Start Chat
                    </button>
                  )}
                </div>
              </div>
            );
          })}
        </div>
      </div>

      {activeChat && (
        <div className="w-96 sticky top-8 h-fit">
          <div className="flex justify-between items-center mb-2 px-2">
            <span className="font-black text-gray-400 text-xs uppercase tracking-widest">Live Session</span>
            <button onClick={() => setActiveChat(null)} className="text-red-500 font-bold text-xs bg-red-50 px-2 py-1 rounded-lg">Close X</button>
          </div>
          <Chat receiverId={activeChat.id} receiverName={activeChat.name} />
        </div>
      )}
    </div>
  );
};

export default RequestsInbox;
